CREATE DATABASE CAMIIGAME;
USE CAMIIGAME;

CREATE TABLE GAMES(
	GAMCOD INT NOT NULL AUTO_INCREMENT,
    GAMNOME VARCHAR(500),
    GAMPLATAFORMA VARCHAR(500),
    GAMDEV VARCHAR(500),
    GAMPUBLISH VARCHAR(500),
    GAMGENERO VARCHAR(500),
    GAMIMG VARCHAR(500),
    PRIMARY KEY(GAMCOD)
);
